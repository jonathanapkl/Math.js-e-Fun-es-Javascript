Mais opções de recursos Úteis de Math.js

📏 Conversão de unidades


        console.log(math.unit('5 cm').to('m').toString());      // 0.05 m
        console.log(math.unit('3 kg').to('g').toString());      // 3000 g

🧮 Simplificação de expressões algébricas


        const simplificada = math.simplify('2x + 3x - x');
        console.log("Expressão simplificada:", simplificada.toString());  // 4x

🧠 Derivada simbólica


        const derivada = math.derivative('x^2 + x', 'x');
        console.log("Derivada:", derivada.toString());  // 2*x + 1

📐 Trigonometria


        console.log("Seno(90°):", math.sin(math.unit(90, 'deg')));  // 1
        console.log("Cosseno(0°):", math.cos(math.unit(0, 'deg'))); // 1

🧪 Avaliação de expressões com variáveis


        console.log(math.evaluate('a * b + c', { a: 2, b: 3, c: 4 })); // 10

📊 Média, mediana e desvio padrão


        const dados = [4, 8, 6, 5, 3];
        console.log("Média:", math.mean(dados));        // 5.2
        console.log("Mediana:", math.median(dados));    // 5
        console.log("Desvio padrão:", math.std(dados)); // 1.720

📏 Formato bonito de resultado


        console.log(math.format(1234567.89, { notation: 'exponential' })); // 1.23456789e+6
        console.log(math.format(0.0001234, { precision: 4 }));             // 0.0001234
