Mais opções de Funções Úteis de Javascript

✅ 1. Manipulação de Strings


        const texto = "JavaScript é poderoso!";
        console.log("Tamanho: ", texto.length);                         // 23
        console.log("Maiúsculas: ", texto.toUpperCase());               // JAVASCRIPT É PODEROSO!
        console.log("Minúsculas: ", texto.toLowerCase());               // javascript é poderoso!
        console.log("Contém 'poderoso'? ", texto.includes("poderoso")); // true
        console.log("Substring: ", texto.substring(0, 10));             // JavaScript

✅ 2. Números e Math


        console.log("Pi: ", Math.PI);                           // 3.1415...
        console.log("Arredondar pra cima: ", Math.ceil(4.2));   // 5
        console.log("Arredondar pra baixo: ", Math.floor(4.9)); // 4
        console.log("Número aleatório: ", Math.random());       // ex: 0.374

✅ 3. Arrays (Vetores)


        const frutas = ["maçã", "banana", "laranja"];
        console.log("Tamanho: ", frutas.length);            // 3
        frutas.push("uva");
        console.log("Adicionando uva: ", frutas);           // ["maçã", "banana", "laranja", "uva"]
        rutas.pop();
        console.log("Removendo último: ", frutas);          // ["maçã", "banana", "laranja"]
        console.log("Primeiro elemento: ", frutas[0]);      // maçã

✅ 4. Objetos


        const usuario = {
          nome: "Jonathan",
          idade: 22,
          ativo: true
        };
        console.log("Nome: ", usuario.nome);          // Jonathan
        usuario.idade = 23;
        console.log("Nova idade: ", usuario.idade);   // 23
        console.log("Está ativo? ", usuario.ativo);   // true

✅ 5. Funções


        function somar(a, b) {
          return a + b;
        }
        console.log("Soma: ", somar(5, 3));   // 8

✅ 6. Laços (for, while)


        for (let i = 1; i <= 3; i++) {
          console.log("Contando: ", i);
        }
        let n = 0;
        while (n < 3) {
          console.log("While: ", n);
          n++;
        }

✅ 7. Condicionais


        let idade = 18;
        if (idade >= 18) {
          console.log("Maior de idade");
        } else {
          console.log("Menor de idade");
        }
